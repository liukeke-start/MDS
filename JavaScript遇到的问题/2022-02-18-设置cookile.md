### 设置cookie24小时后过期
```javascript
function initCloseBanner () {
  // 判断如果没有这个节点，就return
  const adsite = $('.adsite');
  if(!adsite[0]) return;
  $('.adsite .close').on('click', function () {
    adsite.hide();
  });
  if(!document.cookie.includes('adsite_show=')) {
    setCookie();
    adsite.show();
  } else {
    adsite.hide();
  }
}

function setCookie() {
  let d = new Date();
  d.setTime(d.getTime() + 24 * 60 * 60 * 1000);
  document.cookie = `adsite_show=popup-ad;expires=${d.toUTCString()}`;
  let result = document.cookie;
  return result;
}
// 调用函数
initCloseBanner(); 
```
