### 渲染数据的时候，要给其中一个li加类名
```javascript
 // rankingStyle 中的写法
 export const rankingStyle = str => {
  let style = "";
  switch (str) {
    case 1:
      style = "one";
      break;
    case 2:
      style = "two";
      break;
    case 3:
      style = "three";
      break;
  }
  return style;
};

import { rankingStyle } from "./lib/ranking"; // 引入到index.js中
$.ajax({
    url: "",
    dataType: "jsonp",
    data: {}
}).done(({ data }) => {
 $("#app").html(`
  <ul>
    ${data.map(item => {
      const { rank, title, flag, num } = item.attributes;
      return `
        <li>
          <a href="#">
              <span class="${rankingStyle(rank)}">${rank}</span>
          </a>
       </li>
       `;
        }).join("")}
      </ul>
    `);
  });
  });
  //如果不加join的话，渲染出来会有数组中的，
```

### 如果浏览器报不是同源策略的错误
- 可能是渲染数据的时候跨域不对，需要用ajax请求

### 回到顶部
``` javascript
var $toBack = $(".toBack");
let app = {
  goBack: function () {
    $(window).on("scroll", function () {
      // 判断显示还是隐藏
      if ($(this).scrollTop() >= $(this).height()) {
        $toBack.fadeIn("slow");
      } else {
        $toBack.fadeOut("slow");
      }
    });
    $toBack.on("click", function () {
      $("html,body").animate({ scrollTop: "0" }, 500);
    });
  }
};
app.goBack();
```
