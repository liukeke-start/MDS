### jjs中字符串去掉空格
#### replace正则匹配方法
- 去除所有空格:
str = str.replace(/\s+/g,"");
```javascript
// 例子
// 字符串   
let str= ' hao ni ';
str = str.replace(/\s+/g,''); // "haoni"

// 数组
let arr2 = ['ads sdfbo',' sdfsdf  sdf ',' sdfsdsf '];
arr2.forEach(function (value) {
  value = value.replace(/\s+/g,'');
});
console.log(arr2) // ["ads sdfbo", " sdfsdf  sdf ", " sdfsdsf "]

// 对象
let arr3 = {
    name: ' st art ',
    age: ' 21 ',
    code: ' typ e'
}
for( let val in arr3 ){
    arr3[val] = arr3[val].replace(/\s+/g,'');
}
console.log(arr3)  // {name: "start", age: "21", code: "type"}
```
- 去除两头空格:
str = str.replace(/^\s+|\s+$/g,"");
```javascript
// 例子   
let str= ' hao ni ';
str = str.replace(/^\s+|\s+$/g,""); // "hao ni"
```
- 去除左空格：
str=str.replace( /^\s/, '');
```javascript
// 例子   
let str= ' hao ni ';
str = str.replace( /^\s/, ''); // "hao ni "
```
- 去除右空格：
str=str.replace(/(\s$)/g, "");
```javascript
// 例子   
let str= ' hao ni ';
str = str.replace(/\s+/g,''); // "haoni"
```
#### str.trim()方法
- 作用：用来删除字符串两端的空白字符并返回
- 返回值：trim方法并不影响原来的字符串本身，它返回的是一个新的字符串。
- 缺陷：只能去除字符串两端的空格，不能去除中间的空格
```javascript
// 例子   
let str= ' hao ni ';
str = str.trim(); // "hao ni"
```
- jq中则是$.trim(str)
